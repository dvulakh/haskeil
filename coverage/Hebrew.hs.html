   12 class Pretty a where
   13   prettyShow :: a -> String
   14   prettyReadPrec :: String -> Maybe (a, String)
   15   prettyRead :: String -> a
   16   prettyRead s = case prettyReadPrec s of
   17                    Just (a, "") -> a

  156 type HWord = [HLetter]
  157 type HFWord = [HFLetter]

  160 instance Pretty a => Pretty [a] where
  161   prettyShow :: [a] -> String
  162   prettyShow = (>>= prettyShow)
  163   prettyReadPrec :: String -> Maybe ([a], String)
  164   prettyReadPrec = Just . prettyReadPrec'
  165    where
  166     prettyReadPrec' :: Pretty a => String -> ([a], String)
  167     prettyReadPrec' s = maybe
  168       ([], s)
  169       (uncurry ($) . bimap (first . (:)) prettyReadPrec')
  170       (prettyReadPrec s)

  173 finalize :: HLetter -> HFLetter
  174 finalize Chaf    = FFChaf
  175 finalize Mem     = FFMem
  176 finalize Nun     = FFNun
  177 finalize Pay     = FFPay
  178 finalize Tzadi   = FFTzadi
  179 finalize fletter = toFinal fletter

  182 toFinal :: HLetter -> HFLetter
  183 toFinal = toEnum . fromEnum

  186 fromFinal :: HFLetter -> HLetter
  187 fromFinal FFChaf  = Chaf
  188 fromFinal FFMem   = Mem
  189 fromFinal FFNun   = Nun
  190 fromFinal FFPay   = Pay
  191 fromFinal FFTzadi = Tzadi
  192 fromFinal fletter = toEnum $ fromEnum fletter

  239 properWord w = case reverse w of
