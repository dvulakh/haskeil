   43 computeGematria :: Gematria -> HFWord -> Int
   44 computeGematria Hechrachi = stam hechrachiValue
   45 computeGematria Gadol     = stam gadolValue
   46 computeGematria Kattan    = stam $ (+ 1) . (`mod` 9) . fromEnum . removeFinals
   47 computeGematria Siduri    = stam $ (+ 1) . fromEnum
   48 --computeGematria Boneh     = undefined
   49 computeGematria Kidmi     = computeGematria Hechrachi . (enumFromTo FAlef =<<)
   50 computeGematria Prati     = stam $ (^ (2 :: Int)) . hechrachiValue
   51 computeGematria Merubah   = (^ (2 :: Int)) . computeGematria Hechrachi
   52 computeGematria Meshulash = stam $ (^ (3 :: Int)) . hechrachiValue
   53 computeGematria Akhor     = sum . zipWith (*) [1 ..] . map hechrachiValue
   54 --computeGematria Mispari   = undefined
   55 computeGematria BeMilui   = computeGematria Hechrachi . (letterSpelling =<<)
   56 computeGematria Neelam    = computeGematria Hechrachi . (spell =<<)
