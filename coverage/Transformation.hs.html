   15 toHFLetter :: Int -> HFLetter

   17 
   18 fromHLetter :: HLetter -> Int

   24 applyTransformation :: Transformation -> HFWord -> HFWord
   25 applyTransformation Aatat = id
   26 applyTransformation Atbash =
   27   map $ toEnum . (maxLetter -) . fromHLetter . fromFinal
   28 applyTransformation Albam =
   29   map
   30     $ toEnum
   31     . (`mod` maxLetter)

   35 applyTransformation Achbi = map $ \h ->
   36   let (group, position) = fromEnum (removeFinals h) `divMod` halfLetter

   39 applyTransformation AyakBakar =
   40   map
   41     $ toEnum
   42     . (`mod` (1 + fromEnum (maxBound :: HFLetter)))
   43     . (+ 9)

   45 applyTransformation Ofanim    = map $ removeFinals . last . letterSpelling
   46 applyTransformation AkhasBeta = map $ \h -> case fromFinal h of
   47   Tav -> FTav
   48   _ ->
   49     let (group, position) = fromEnum (fromFinal h) `divMod` 7

   51 applyTransformation Avgad =
