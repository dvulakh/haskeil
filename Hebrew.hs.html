  163 instance Pretty a => Pretty [a] where
  164   prettyShow :: [a] -> String
  165   prettyShow = (>>= prettyShow)
  166   prettyReadPrec :: String -> Maybe ([a], String)
  167   prettyReadPrec = Just . prettyReadPrec'
  168    where
  169     prettyReadPrec' :: Pretty a => String -> ([a], String)
  170     prettyReadPrec' s =
  171       fromMaybe ([], s)
  172         $   uncurry ($)
  173         <$> bimap (first . (:)) prettyReadPrec'
  174         <$> prettyReadPrec s
